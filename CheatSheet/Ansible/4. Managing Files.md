- The File to be copied can be in **same folder** or **under folder name** `files`, you need not specify path, name is enough.

`install.yml` playbook
```yml
---
- hosts: web_servers
  become: true
  tasks:
    - name: copy html file for site
      tags: apache,apache,apache2,httpd
      copy:
        src: default_site.html                # File path to be copied
        dest: /var/www/html/index.html        # Destination path to be copied
        owner: root                           # owner of the copied file
        group: root                           # group of the copied file
        mode: 644                             # permission for the file
```

```yml
- hosts: workstations
   become: true
   tasks:

   - name: install unzip
     package:
       name: unzip
 
   - name: install terraform
     unarchive:                     # To unzip the zip file
      src: https://releases.hashicorp.com/terraform/0.12.28/terraform_0.12.28_linux_amd64.zip                                   # URL to Zip file
       dest: /usr/local/bin         # Destination for the unziped file
       remote_src: yes              # Download file from internet
       mode: 0755
       owner: root
       group: root
```
