`site.yml`
```yml
---

- hosts: all                        # will run across all hosts
  become: true
  pre_tasks:                        # Mandates this task will run first

  - name: install updates (CentOS)
    tags: always                   # tag 'always' runs, when playbook is run by tag
    dnf:
      update_only: yes             # upgrade distro
      update_cache: yes
    changed_when: false            # don't log dit update has a change
    when: ansible_distribution == "CentOS"

  - name: install updates (Ubuntu)
    tags: always
    apt:
      upgrade: dist               # upgrade distro
      update_cache: yes
    when: ansible_distribution == "Ubuntu"


- hosts: web_servers              # will only run on [web_server] ip
  become: true
  tasks:

  - name: install apache and php for Ubuntu servers
    tags: apache,apache2,ubuntu   # only tagged task will run, with ansible tagged command
    apt:
      name:
        - apache2
        - libapache2-mod-php
      state: latest
    when: ansible_distribution == "Ubuntu"

  - name: install apache and php for CentOS servers
    tags: apache,centos,httpd
    dnf:
      name:
        - httpd
        - php
      state: latest
    when: ansible_distribution == "CentOS"


- hosts: db_servers            # will only run on [db_servers] ip
  become: true
  tasks:

  - name: install mariadb package (CentOS)
    tags: centos,db,mariadb
    dnf:
      name: mariadb
      state: latest
    when: ansible_distribution == "CentOS"

  - name: install mariadb package (Ubuntu)
    tags: db,mariadb,ubuntu
    apt:
      name: mariadb-server
      state: latest
    when: ansible_distribution == "Ubuntu"

- hosts: file_servers        # will only run on [file_servers] ip
  become: true
  tasks:

  - name: install samba package
    tags: samba
    package:
      name: samba
      state: latest
```

### List the available tags in a playbook

```sh
ansible-playbook --list-tags playbook_with_tags.yml
```

```sh
ansible-playbook --list-tags site.yml
```

### Examples of running a playbook but targeting specific tags

-  Tag `always` runs, when playbook is run by tag
- Only tagged task will run, with ansible tagged command
```
 ansible-playbook --tags db --ask-become-pass site_with_tags.yml
 ansible-playbook --tags centos --ask-become-pass site_with_tags.yml
 ansible-playbook --tags "apache db" --ask-become-pass site_with_tags.yml
```
