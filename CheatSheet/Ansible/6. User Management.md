
```yml
 - hosts: all
   become: true
   tasks:

   - name: create simone user
     tags: always
     user:
       name: simone      # user to be created
       groups: root      # user to be added to group root
       
   - name: add ssh key for simone
     tags: always
     authorized_key:    # module to add ssh.pub key
       user: simone     # user for who ssh.pub needs to be added
       key: "ssh-ed25519 AAAAC3NzaC1lZDI1NTE5AAAAIAe7/ofWLNBq3+fRn3UmgAizdicLs9vcS4Oj8VSOD1S/ ansible"  # ssh.pub key

  - name: add sudoers file for simone
     tags: always
     copy:
       src: sudoer_simone    # copy file to path to give access to sudo privilages
       dest: /etc/sudoers.d/simone
       owner: root
       group: root
       mode: 0440
```

other ways to copy keys.
```yml
- name: Set authorized key taken from file
  ansible.posix.authorized_key:
    user: charlie
    state: present
    key: "{{ lookup('file', '/home/charlie/.ssh/id_rsa.pub') }}"

- name: Set authorized keys taken from url
  ansible.posix.authorized_key:
    user: charlie
    state: present
    key: https://github.com/charlie.keys
```

`sudoer_simone` file
```
 simone ALL=(ALL) NOPASSWD: ALL
```

edit `asible.cfg` add user to be used to login as sudo, `--ask-become-pass` is not required in command, to run playbook after this.
```
remote_user = simone
```

- **Create a `bootstrap.yml` to set the user for ansible and initial settings to be provisioned.**
`bootstrap.yml`
```yml
---
 
 - hosts: all
   become: true
   pre_tasks:
 
   - name: install updates (CentOS)
     tags: always
     dnf:
       update_only: yes
       update_cache: yes
     when: ansible_distribution == "CentOS"
 
   - name: install updates (Ubuntu)
     tags: always
     apt:
       upgrade: dist
       update_cache: yes
     when: ansible_distribution == "Ubuntu"
 
 - hosts: all
   become: true
   tasks:
 
   - name: create simone user
     user:
       name: simone
       groups: root
 
   - name: add ssh key for simone
     authorized_key:
       user: simone
       key: "ssh-ed25519 AAAAC3NzaC1lZDI1NTE5AAAAIAe7/ofWLNBq3+fRn3UmgAizdicLs9vcS4Oj8VSOD1S/ ansible"
      
   - name: add sudoers file for simone
     copy:
       src: sudoer_simone
       dest: /etc/sudoers.d/simone
       owner: root
       group: root
       mode: 0440
```
